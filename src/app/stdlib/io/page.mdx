    enum Level {
        UHC_LOG_TRACE,
        UHC_LOG_DEBUG,
        UHC_LOG_INFO,
        UHC_LOG_WARN,
        UHC_LOG_ERROR,
    };
    void trace(const char* fmt, ...);
    void debug(const char* fmt, ...);
    void info (const char* fmt, ...);
    void warn (const char* fmt, ...);
    void error(const char* fmt, ...);
    void msg  (const char* fmt, ...);

namespace File {

        FILE* handle;
        long long size;
