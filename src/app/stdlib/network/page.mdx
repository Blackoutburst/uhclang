    #if defined(_WIN32) || defined(_WIN64)
      struct It {
        SOCKET sockfd;
        HANDLE thread;
      };
    #else
      struct It {
        int32_t sockfd;
        pthread_t thread;
      };
    #endif
    
      TcpClient::It create(char* ip, int16_t port);
      void destroy(TcpClient::It& client);

      unsigned char read(TcpClient::It& client, char* buffer, uint32_t size);
      unsigned char readAll(TcpClient::It& client, char* buffer, uint32_t size);
    }

  namespace TcpServer {
    #if defined(_WIN32) || defined(_WIN64)
      struct It {
        SOCKET sockfd;
        unsigned char running;
        uint16_t slots;
      };
    #else
      struct It {
        int32_t sockfd;
        unsigned char running;
        uint16_t slots;
      };
    #endif

    TcpServer::It create(int16_t port, uint16_t slots);
    void destroy(TcpServer::It& server);
    void listen(TcpServer::It& server, void (*callback)(TcpClient::It*));
    unsigned char write(TcpClient::It& client, char* buffer, uint32_t size);
    unsigned char read(TcpClient::It& client, char* buffer, uint32_t size);
    unsigned char readAll(TcpClient::It& client, char* buffer, uint32_t size);
    void broadcast(List::It<TcpClient::It> clients, char* buffer, uint32_t size);
  }
