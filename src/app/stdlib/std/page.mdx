# STD

Various utilities such as mutex, thread, ...

- [Buffer](/stdlib/std/buffer) — Binary buffer with cursor-based reading and writing
- [DynamicLibrary](/stdlib/std/dynamiclibrary) — Cross-platform shared library loader
- [List](/stdlib/std/list) — Generic doubly-linked list
- [Mutex](/stdlib/std/mutex) — Mutual exclusion primitive for thread synchronization
- [Thread](/stdlib/std/thread) — Lightweight thread creation and joining

namespace Mutex {
    #if defined(_WIN32) || defined(_WIN64)

        pthread_mutex_t handle
      }

namespace Buffer {
    struct It {
        U0* handle
        U8 order
        U32 cursor
    };

namespace Thread {
    typedef U0* (*Fn)(U0*)
    #if defined(_WIN32) || defined(_WIN64)
      struct It {
        HANDLE handle
      }
    #else
      struct It {
        pthread_t handle
      }
    #endif
}

namespace DynamicLibrary {
    #if defined(_WIN32) || defined(_WIN64)
      struct It {
        HINSTANCE handle
        I8* name
        U8 isValid
      }
    #else
      struct It {
        U0* handle
        I8* name
        U8 isValid
      }
    #endif
}

namespace List {
    struct It {
        std::list<T> handle
    };
